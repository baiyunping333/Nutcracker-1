00986    template<class Predicate>
00987    void sort(Predicate p)
00988    {
00989       if(node_traits::get_next(this->get_root_node()) 
00990          != node_traits::get_previous(this->get_root_node())){
00991          list_impl carry(this->get_value_traits());
00992          detail::array_initializer<list_impl, 64> counter(this->get_value_traits());
00993          int fill = 0;
00994          while(!this->empty()){
00995             carry.splice(carry.cbegin(), *this, this->cbegin());
00996             int i = 0;
00997             while(i < fill && !counter[i].empty()) {
00998                counter[i].merge(carry, p);
00999                carry.swap(counter[i++]);
01000             }
01001             carry.swap(counter[i]);
01002             if(i == fill)
01003                ++fill;
01004          }
01005          for (int i = 1; i < fill; ++i)
01006             counter[i].merge(counter[i-1], p);
01007          this->swap(counter[fill-1]);
01008       }
01009    }

01039    template<class Predicate>
01040    void merge(list_impl& x, Predicate p)
01041    {
01042       const_iterator e(this->cend()), ex(x.cend());
01043       const_iterator b(this->cbegin());
01044       while(!x.empty()){
01045          const_iterator ix(x.cbegin());
01046          while (b != e && !p(*ix, *b)){
01047             ++b;
01048          }
01049          if(b == e){
01050             //Now transfer the rest to the end of the container
01051             this->splice(e, x);
01052             break;
01053          }
01054          else{
01055             size_type n(0);
01056             do{
01057                ++ix; ++n;
01058             } while(ix != ex && p(*ix, *b));
01059             this->splice(b, x, x.begin(), ix, n);
01060          }
01061       }
01062    }
