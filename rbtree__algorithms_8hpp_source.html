00403    {

00785    static void rebalance_after_erasure(node_ptr header, node_ptr x, node_ptr x_parent)
00786    {
00787       while(x != NodeTraits::get_parent(header) && (x == 0 || NodeTraits::get_color(x) == NodeTraits::black())){
00788          if(x == NodeTraits::get_left(x_parent)){
00789             node_ptr w = NodeTraits::get_right(x_parent);
00790             if(NodeTraits::get_color(w) == NodeTraits::red()){
00791                NodeTraits::set_color(w, NodeTraits::black());
00792                NodeTraits::set_color(x_parent, NodeTraits::red());
00793                tree_algorithms::rotate_left(x_parent, header);
00794                w = NodeTraits::get_right(x_parent);
00795             }
00796             if((NodeTraits::get_left(w) == 0 || NodeTraits::get_color(NodeTraits::get_left(w))  == NodeTraits::black()) &&
00797                (NodeTraits::get_right(w) == 0 || NodeTraits::get_color(NodeTraits::get_right(w)) == NodeTraits::black())){
00798                NodeTraits::set_color(w, NodeTraits::red());
00799                x = x_parent;
00800                x_parent = NodeTraits::get_parent(x_parent);
00801             } 
00802             else {
00803                if(NodeTraits::get_right(w) == 0 || NodeTraits::get_color(NodeTraits::get_right(w)) == NodeTraits::black()){
00804                   NodeTraits::set_color(NodeTraits::get_left(w), NodeTraits::black());
00805                   NodeTraits::set_color(w, NodeTraits::red());
00806                   tree_algorithms::rotate_right(w, header);
00807                   w = NodeTraits::get_right(x_parent);
00808                }
00809                NodeTraits::set_color(w, NodeTraits::get_color(x_parent));
00810                NodeTraits::set_color(x_parent, NodeTraits::black());
00811                if(NodeTraits::get_right(w))
00812                   NodeTraits::set_color(NodeTraits::get_right(w), NodeTraits::black());
00813                tree_algorithms::rotate_left(x_parent, header);
00814                break;
00815             }
00816          }
00817          else {
00818             // same as above, with right_ <-> left_.
00819             node_ptr w = NodeTraits::get_left(x_parent);
00820             if(NodeTraits::get_color(w) == NodeTraits::red()){
00821                NodeTraits::set_color(w, NodeTraits::black());
00822                NodeTraits::set_color(x_parent, NodeTraits::red());
00823                tree_algorithms::rotate_right(x_parent, header);
00824                w = NodeTraits::get_left(x_parent);
00825             }
00826             if((NodeTraits::get_right(w) == 0 || NodeTraits::get_color(NodeTraits::get_right(w)) == NodeTraits::black()) &&
00827                (NodeTraits::get_left(w) == 0 || NodeTraits::get_color(NodeTraits::get_left(w)) == NodeTraits::black())){
00828                NodeTraits::set_color(w, NodeTraits::red());
00829                x = x_parent;
00830                x_parent = NodeTraits::get_parent(x_parent);
00831             }
00832             else {
00833                if(NodeTraits::get_left(w) == 0 || NodeTraits::get_color(NodeTraits::get_left(w)) == NodeTraits::black()){
00834                   NodeTraits::set_color(NodeTraits::get_right(w), NodeTraits::black());
00835                   NodeTraits::set_color(w, NodeTraits::red());
00836                   tree_algorithms::rotate_left(w, header);
00837                   w = NodeTraits::get_left(x_parent);
00838                }
00839                NodeTraits::set_color(w, NodeTraits::get_color(x_parent));
00840                NodeTraits::set_color(x_parent, NodeTraits::black());
00841                if(NodeTraits::get_left(w))
00842                   NodeTraits::set_color(NodeTraits::get_left(w), NodeTraits::black());
00843                tree_algorithms::rotate_right(x_parent, header);
00844                break;
00845             }
00846          }

00853    static void rebalance_after_insertion(node_ptr header, node_ptr p)
00854    {
00855       NodeTraits::set_color(p, NodeTraits::red());
00856       while(p != NodeTraits::get_parent(header) && NodeTraits::get_color(NodeTraits::get_parent(p)) == NodeTraits::red()){
00857          node_ptr p_parent(NodeTraits::get_parent(p));
00858          node_ptr p_parent_parent(NodeTraits::get_parent(p_parent));
00859          if(tree_algorithms::is_left_child(p_parent)){
00860             node_ptr x = NodeTraits::get_right(p_parent_parent);
00861             if(x && NodeTraits::get_color(x) == NodeTraits::red()){
00862                NodeTraits::set_color(p_parent, NodeTraits::black());
00863                NodeTraits::set_color(p_parent_parent, NodeTraits::red());
00864                NodeTraits::set_color(x, NodeTraits::black());
00865                p = p_parent_parent;
00866             }
00867             else {
00868                if(!tree_algorithms::is_left_child(p)){
00869                   p = p_parent;
00870                   tree_algorithms::rotate_left(p, header);
00871                }
00872                node_ptr new_p_parent(NodeTraits::get_parent(p));
00873                node_ptr new_p_parent_parent(NodeTraits::get_parent(new_p_parent));
00874                NodeTraits::set_color(new_p_parent, NodeTraits::black());
00875                NodeTraits::set_color(new_p_parent_parent, NodeTraits::red());
00876                tree_algorithms::rotate_right(new_p_parent_parent, header);
00877             }
00878          }
00879          else{
00880             node_ptr x = NodeTraits::get_left(p_parent_parent);
00881             if(x && NodeTraits::get_color(x) == NodeTraits::red()){
00882                NodeTraits::set_color(p_parent, NodeTraits::black());
00883                NodeTraits::set_color(p_parent_parent, NodeTraits::red());
00884                NodeTraits::set_color(x, NodeTraits::black());
00885                p = p_parent_parent;
00886             }
00887             else{
00888                if(tree_algorithms::is_left_child(p)){
00889                   p = p_parent;
00890                   tree_algorithms::rotate_right(p, header);
00891                }
00892                node_ptr new_p_parent(NodeTraits::get_parent(p));
00893                node_ptr new_p_parent_parent(NodeTraits::get_parent(new_p_parent));
00894                NodeTraits::set_color(new_p_parent, NodeTraits::black());
00895                NodeTraits::set_color(new_p_parent_parent, NodeTraits::red());
00896                tree_algorithms::rotate_left(new_p_parent_parent, header);
00897             }
00898          }
00899       }
00900       NodeTraits::set_color(NodeTraits::get_parent(header), NodeTraits::black());
00901    }
