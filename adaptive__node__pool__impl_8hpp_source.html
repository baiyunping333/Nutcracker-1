00054    struct block_info_t
00055       :  
00056          public hdr_offset_holder,
00057          public multiset_hook_t
00058    {
00059       typedef typename node_slist<void_pointer>::node_slist_t free_nodes_t;
00060       //An intrusive list of free node from this block
00061       free_nodes_t free_nodes;
00062       friend bool operator <(const block_info_t &l, const block_info_t &r)
00063       {
00064 //      {  return l.free_nodes.size() < r.free_nodes.size();   }
00065          //Let's order blocks first by free nodes and then by address
00066          //so that highest address fully free blocks are deallocated.
00067          //This improves returning memory to the OS (trimming).
00068          const bool is_less  = l.free_nodes.size() < r.free_nodes.size();
00069          const bool is_equal = l.free_nodes.size() == r.free_nodes.size();
00070          return is_less || (is_equal && (&l < &r));
00071       }
00072    };
00073    typedef typename bi::make_multiset
00074       <block_info_t, bi::base_hook<multiset_hook_t> >::type  block_multiset_t;

00078 inline std::size_t calculate_alignment
00079    ( std::size_t overhead_percent, std::size_t real_node_size
00080    , std::size_t hdr_size, std::size_t hdr_offset_size, std::size_t payload_per_allocation)
00081 {
00082    //to-do: handle real_node_size != node_size
00083    const std::size_t divisor  = overhead_percent*real_node_size;
00084    const std::size_t dividend = hdr_offset_size*100;
00085    std::size_t elements_per_subblock = (dividend - 1)/divisor + 1;
00086    std::size_t candidate_power_of_2 = 
00087       upper_power_of_2(elements_per_subblock*real_node_size + hdr_offset_size);
00088    bool overhead_satisfied = false;
00089    //Now calculate the wors-case overhead for a subblock
00090    const std::size_t max_subblock_overhead  = hdr_size + payload_per_allocation;
00091    while(!overhead_satisfied){
00092       elements_per_subblock = (candidate_power_of_2 - max_subblock_overhead)/real_node_size;
00093       const std::size_t overhead_size = candidate_power_of_2 - elements_per_subblock*real_node_size;
00094       if(overhead_size*100/candidate_power_of_2 < overhead_percent){
00095          overhead_satisfied = true;
00096       }
00097       else{
00098          candidate_power_of_2 <<= 1;
00099       }
00100    }
00101    return candidate_power_of_2;
00102 }

00104 inline void calculate_num_subblocks
00105    (std::size_t alignment, std::size_t real_node_size, std::size_t elements_per_block
00106    , std::size_t &num_subblocks, std::size_t &real_num_node, std::size_t overhead_percent
00107    , std::size_t hdr_size, std::size_t hdr_offset_size, std::size_t payload_per_allocation)
00108 {
00109    std::size_t elements_per_subblock = (alignment - hdr_offset_size)/real_node_size;
00110    std::size_t possible_num_subblock = (elements_per_block - 1)/elements_per_subblock + 1;
00111    std::size_t hdr_subblock_elements = (alignment - hdr_size - payload_per_allocation)/real_node_size;
00112    while(((possible_num_subblock-1)*elements_per_subblock + hdr_subblock_elements) < elements_per_block){
00113       ++possible_num_subblock;
00114    }
00115    elements_per_subblock = (alignment - hdr_offset_size)/real_node_size;
00116    bool overhead_satisfied = false;
00117    while(!overhead_satisfied){
00118       const std::size_t total_data = (elements_per_subblock*(possible_num_subblock-1) + hdr_subblock_elements)*real_node_size;
00119       const std::size_t total_size = alignment*possible_num_subblock;
00120       if((total_size - total_data)*100/total_size < overhead_percent){
00121          overhead_satisfied = true;
00122       }
00123       else{
00124          ++possible_num_subblock;
00125       }
00126    }
00127    num_subblocks = possible_num_subblock;
00128    real_num_node = (possible_num_subblock-1)*elements_per_subblock + hdr_subblock_elements;
00129 }

00243    multiallocation_chain allocate_nodes(const std::size_t n)
00244    {
00245       multiallocation_chain chain;
00246       std::size_t i = 0;
00247       try{
00248          priv_invariants();
00249          while(i != n){
00250             //If there are no free nodes we allocate all needed blocks
00251             if (m_block_multiset.empty()){
00252                priv_alloc_block(((n - i) - 1)/m_real_num_node + 1);
00253             }
00254             free_nodes_t &free_nodes = m_block_multiset.begin()->free_nodes;
00255             const std::size_t free_nodes_count_before = free_nodes.size();
00256             if(free_nodes_count_before == m_real_num_node){
00257                --m_totally_free_blocks;
00258             }
00259             const std::size_t num_elems = ((n-i) < free_nodes_count_before) ? (n-i) : free_nodes_count_before;
00260             for(std::size_t j = 0; j != num_elems; ++j){
00261                void *new_node = &free_nodes.front();
00262                free_nodes.pop_front();
00263                chain.push_back(new_node);
00264             }
00265 
00266             if(free_nodes.empty()){
00267                m_block_multiset.erase(m_block_multiset.begin());
00268             }
00269             i += num_elems;
00270          }
00271       }
00272       catch(...){
00273          this->deallocate_nodes(BOOST_CONTAINER_MOVE_NAMESPACE::move(chain));
00274          throw;
00275       }
00276       priv_invariants();
00277       return BOOST_CONTAINER_MOVE_NAMESPACE::move(chain);
00278    }

00320    {
00321       priv_invariants();
00322       //Now check if we've reached the free nodes limit
00323       //and check if we have free blocks. If so, deallocate as much
00324       //as we can to stay below the limit
00325       for( block_iterator itend = m_block_multiset.end()
00326          ; m_totally_free_blocks > max_free_blocks

00337    void priv_reinsert_nodes_in_block(multiallocation_chain &chain, std::size_t n)
00338    {
00339       block_iterator block_it(m_block_multiset.end());
00340       while(n--){
00341          void *pElem = containers_detail::get_pointer(chain.front());
00342          chain.pop_front();
00343          priv_invariants();
00344          block_info_t *block_info = this->priv_block_from_node(pElem);
00345          assert(block_info->free_nodes.size() < m_real_num_node);
00346          //We put the node at the beginning of the free node list
00347          node_t * to_deallocate = static_cast<node_t*>(pElem);
00348          block_info->free_nodes.push_front(*to_deallocate);
00349 
00350          block_iterator this_block(block_multiset_t::s_iterator_to(*block_info));
00351          block_iterator next_block(this_block);
00352          ++next_block;
00353 
00354          //Cache the free nodes from the block
00355          std::size_t this_block_free_nodes = this_block->free_nodes.size();
00356 
00357          if(this_block_free_nodes == 1){
00358             m_block_multiset.insert(m_block_multiset.begin(), *block_info);
00359          }
00360          else{
00361             block_iterator next_block(this_block);
00362             ++next_block;
00363             if(next_block != block_it){
00364                std::size_t next_free_nodes = next_block->free_nodes.size();
00365                if(this_block_free_nodes > next_free_nodes){
00366                   //Now move the block to the new position
00367                   m_block_multiset.erase(this_block);
00368                   m_block_multiset.insert(*block_info);
00369                }
00370             }
00371          }
00372          //Update free block count
00373          if(this_block_free_nodes == m_real_num_node){
00374             ++m_totally_free_blocks;
00375          }
00376          priv_invariants();
00377       }
00378    }

00529    hdr_offset_holder *priv_first_subblock_from_block(block_info_t *block) const
00530    {
