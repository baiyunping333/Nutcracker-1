00143 FwdIt uninitialized_copy_n_dispatch

00146     FwdIt dest, containers_detail::bool_<false>)
00147 {
00148    typedef typename std::iterator_traits<FwdIt>::value_type value_type;
00149    //Save initial destination position
00150    FwdIt dest_init = dest;
00151    typename std::iterator_traits<InIt>::difference_type new_count = count+1;

00153    BOOST_TRY{
00154       //Try to build objects
00155       for (; --new_count; ++dest, ++first){
00156          construct_in_place(containers_detail::get_pointer(&*dest), first);

00158    }

00160       //Call destructors

00187 // uninitialized_copy_copy
00188 // Copies [first1, last1) into [result, result + (last1 - first1)), and
00189 // copies [first2, last2) into
00190 // [result + (last1 - first1), result + (last1 - first1) + (last2 - first2)).
00191 template <class InpIt1, class InpIt2, class FwdIt>
00192 FwdIt uninitialized_copy_copy
00193    (InpIt1 first1, InpIt1 last1, InpIt2 first2, InpIt2 last2, FwdIt result)
00194 {
00195    typedef typename std::iterator_traits<FwdIt>::value_type value_type;
00196    FwdIt mid = std::uninitialized_copy(first1, last1, result);
00197    BOOST_TRY {
00198       return std::uninitialized_copy(first2, last2, mid);
00199    }
00200    BOOST_CATCH(...){
00201       for(;result != mid; ++result){
00202          containers_detail::get_pointer(&*result)->~value_type();
00203       }
00204       BOOST_RETHROW
00205    }
00206    BOOST_CATCH_END
00207 }
