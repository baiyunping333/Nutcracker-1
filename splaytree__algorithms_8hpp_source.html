00067                      , node_ptr leftmost           , node_ptr rightmost)
00068       : pcur_subtree_(pcur_subtree)  , header_(header)
00069       , leftmost_(leftmost)   , rightmost_(rightmost)

00076    {
00077       if(pcur_subtree_){
00078          //Exception can only be thrown by comp, but
00079          //tree invariants still hold. *pcur_subtree is the current root
00080          //so link it to the header.

00083          //Recover leftmost/rightmost pointers
00084          NodeTraits::set_left (header_, leftmost_);
00085          NodeTraits::set_right(header_, rightmost_);
00086       }
00087    }

00717    // bottom-up splay, use data_ as parent for n    | complexity : logarithmic    | exception : nothrow
00718    static void splay_up(node_ptr n, node_ptr header)
00719    {
00720       if(n == header){  // do a splay for the right most node instead
00721          // this is to boost performance of equal_range/count on equivalent containers in the case
00722          // where there are many equal elements at the end
00723          n = NodeTraits::get_right(header);
00724       }
00725 
00726       node_ptr t = header;
00727 
00728       if( n == t ) return;
00729       
00730       for( ;; ){
00731          node_ptr p(NodeTraits::get_parent(n));
00732          node_ptr g(NodeTraits::get_parent(p));
00733 
00734          if( p == t )   break;
00735          
00736          if( g == t ){
00737             // zig
00738             rotate(n);
00739          }
00740          else if ((NodeTraits::get_left(p) == n && NodeTraits::get_left(g) == p)    ||
00741                   (NodeTraits::get_right(p) == n && NodeTraits::get_right(g) == p)  ){
00742             // zig-zig
00743             rotate(p);
00744             rotate(n);
00745          }
00746          else{
00747             // zig-zag
00748             rotate(n);
00749             rotate(n);
00750          }
00751       }
00752    }

00754    // top-down splay | complexity : logarithmic    | exception : strong, note A
00755    template<class KeyType, class KeyNodePtrCompare>
00756    static node_ptr splay_down(node_ptr header, const KeyType &key, KeyNodePtrCompare comp)
00757    {
00758       if(!NodeTraits::get_parent(header))
00759          return header;
00760       //Most splay tree implementations use a dummy/null node to implement.
00761       //this function. This has some problems for a generic library like Intrusive:
00762       //
00763       // * The node might not have a default constructor.
00764       // * The default constructor could throw.
00765       //
00766       //We already have a header node. Leftmost and rightmost nodes of the tree
00767       //are not changed when splaying (because the invariants of the tree don't
00768       //change) We can back up them, use the header as the null node and
00769       //reassign old values after the function has been completed.
00770       node_ptr t = NodeTraits::get_parent(header);
00771       //Check if tree has a single node
00772       if(!NodeTraits::get_left(t) && !NodeTraits::get_right(t))
00773          return t;
00774       //Backup leftmost/rightmost
00775       node_ptr leftmost (NodeTraits::get_left(header));
00776       node_ptr rightmost(NodeTraits::get_right(header));
00777       {
00778          detail::splaydown_rollback<NodeTraits> rollback(&t, header, leftmost, rightmost);
00779          node_ptr null = header;
00780          node_ptr l = null;
00781          node_ptr r = null;
00782 
00783          for( ;; ){
00784             if(comp(key, t)){
00785                if(NodeTraits::get_left(t) == 0 )
00786                   break;
00787                if(comp(key, NodeTraits::get_left(t))){
00788                   t = tree_algorithms::rotate_right(t);
00789 
00790                   if(NodeTraits::get_left(t) == 0)
00791                      break;
00792                   link_right(t, r);
00793                }
00794                else if(comp(NodeTraits::get_left(t), key)){
00795                   link_right(t, r);
00796 
00797                   if(NodeTraits::get_right(t) == 0 )
00798                      break;
00799                   link_left(t, l);
00800                }
00801                else{
00802                   link_right(t, r);
00803                }
00804             }
00805             else if(comp(t, key)){
00806                if(NodeTraits::get_right(t) == 0 )
00807                   break;
00808 
00809                if(comp(NodeTraits::get_right(t), key)){
00810                      t = tree_algorithms::rotate_left( t );
00811 
00812                      if(NodeTraits::get_right(t) == 0 )
00813                         break;
00814                      link_left(t, l);
00815                }
00816                else if(comp(key, NodeTraits::get_right(t))){
00817                   link_left(t, l);
00818 
00819                   if(NodeTraits::get_left(t) == 0)
00820                      break;
00821 
00822                   link_right(t, r);
00823                }
00824                else{
00825                   link_left(t, l);
00826                }
00827             }
00828             else{
00829                break;
00830             }
00831          }
00832 
00833          assemble(t, l, r, null);
00834          rollback.release();
00835       }
00836 
00837       //t is the current root
00838       NodeTraits::set_parent(header, t);
00839       NodeTraits::set_parent(t, header);
00840       //Recover leftmost/rightmost pointers
00841       NodeTraits::set_left (header, leftmost);
00842       NodeTraits::set_right(header, rightmost);
00843       return t;
00844    }

00927    // rotate n with its parent                     | complexity : constant    | exception : nothrow
00928    static void rotate(node_ptr n)
00929    {
00930       node_ptr p = NodeTraits::get_parent(n);
00931       node_ptr g = NodeTraits::get_parent(p);
00932       //Test if g is header before breaking tree 
00933       //invariants that would make is_header invalid
00934       bool g_is_header = is_header(g);
00935       
00936       if(NodeTraits::get_left(p) == n){
00937          NodeTraits::set_left(p, NodeTraits::get_right(n));
00938          if(NodeTraits::get_left(p) != 0)
00939             NodeTraits::set_parent(NodeTraits::get_left(p), p);
00940          NodeTraits::set_right(n, p);
00941       }
00942       else{ // must be ( p->right == n )
00943          NodeTraits::set_right(p, NodeTraits::get_left(n));
00944          if(NodeTraits::get_right(p) != 0)
00945             NodeTraits::set_parent(NodeTraits::get_right(p), p);
00946          NodeTraits::set_left(n, p);
00947       }
00948 
00949       NodeTraits::set_parent(p, n);
00950       NodeTraits::set_parent(n, g);
00951 
00952       if(g_is_header){
00953          if(NodeTraits::get_parent(g) == p)
00954             NodeTraits::set_parent(g, n);
00955          else{//must be ( g->right == p )
00956             BOOST_INTRUSIVE_INVARIANT_ASSERT(0);
00957             NodeTraits::set_right(g, n);
00958          }
00959       }
00960       else{
00961          if(NodeTraits::get_left(g) == p)
00962             NodeTraits::set_left(g, n);
00963          else  //must be ( g->right == p )
00964             NodeTraits::set_right(g, n);
00965       }
00966    }
