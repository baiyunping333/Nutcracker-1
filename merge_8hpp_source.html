00020 // Merge two SSE as: seq as specified, sign of 1 followed by sign of 2, err from
00021 // either 1 or 2 with non empty err.
00022 #define BOOST_DETAIL_PP_SIGN_AUX_PARSER_MERGE( \
00023         seq_sign_err1, seq_sign_err2, seq) \
00024     ( /* remaining seq */ \
00025         seq \
00026     , /* sign 1 followed by 2 */ \
00027         BOOST_DETAIL_PP_SIGN_PARSER_SIGN(seq_sign_err1) /* seq append */ \
00028         /* `TAIL()` removes leading `(NIL)` (to handle empty seq) */ \
00029         BOOST_PP_SEQ_TAIL(BOOST_DETAIL_PP_SIGN_PARSER_SIGN(seq_sign_err2)) \
00030     , /* err from either one that is not OK (or NONE if both OK) */ \
00031         BOOST_DETAIL_PP_SIGN_PARSER_ERR( \
00032             BOOST_PP_IIF(BOOST_DETAIL_PP_SIGN_PARSER_HAS_ERR( \
00033                     seq_sign_err1), \
00034                 seq_sign_err1 /* take err1 (because not OK) */ \
00035             , \
00036                 seq_sign_err2 /* take err2 (might or not be OK) */ \
00037             ) \
00038         ) \
00039     )
