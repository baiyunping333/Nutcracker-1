00022 // First applies common validation checks, if they pass apply additional
00023 // validation checks as defined by the user in `validation_macro(seq_sign_err)`.
00024 #define BOOST_DETAIL_PP_SIGN_VALIDATE_BASIC( \
00025         seq_sign_err, sign_count, validation_macro) \
00026     /* common validation checks */ \
00027     /* forward existing error (if any) */ \
00028     BOOST_PP_IIF(BOOST_DETAIL_PP_SIGN_PARSER_HAS_ERR(seq_sign_err), \
00029         seq_sign_err \
00030         BOOST_PP_TUPLE_EAT(1) \
00031     /* when done, only seq elem left should be NIL (size != 1 invalid) */ \
00032     , BOOST_PP_IIF(BOOST_PP_NOT_EQUAL(BOOST_PP_SEQ_SIZE( \
00033             BOOST_DETAIL_PP_SIGN_PARSED_SEQ(seq_sign_err)), 1), \
00034         BOOST_DETAIL_PP_SIGN_VALIDATE_RETURN_ERR(seq_sign_err, \
00035                 ERROR_unable_to_parse_all_tokens_because_of_invalid_syntax_or_order) \
00036         BOOST_PP_TUPLE_EAT(1) \
00037     /* when done, exactly ID_MAX elems in sign */ \
00038     , BOOST_PP_IIF(BOOST_PP_NOT_EQUAL( \
00039             BOOST_DETAIL_PP_SIGN_AUX_PARSED_COUNT(seq_sign_err), \
00040             sign_count), \
00041         BOOST_DETAIL_PP_SIGN_VALIDATE_RETURN_ERR(seq_sign_err, \
00042                 ERROR_missing_one_or_more_signature_tokens) \
00043         BOOST_PP_TUPLE_EAT(1) \
00044     , /* user defined validation checks */ \
00045         validation_macro \
00046     )))(seq_sign_err)
