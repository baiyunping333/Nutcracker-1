00022 // Utilities to handle sequences starting with `(NIL)` (to support empty seq):
00023 //     nilseq := (BOOST_PP_NIL) [(elem0) (elem1) ...]
00024 
00025 // NOTE: `BOOST_PP_LIST` serves a similar purpose but it could not be used in
00026 // order to reuse signature parsing code. For example, parsed parameters are
00027 // represented as NILSEQ so parsers code used to parse sign sequence (wich
00028 // operate on SEQ) can be reused. Similarly, for initializers, concepts, etc.
00029 
00030 #define BOOST_DETAIL_PP_NILSEQ_SIZE(nilseq) \
00031     BOOST_PP_DEC(BOOST_PP_SEQ_SIZE(nilseq)) /* DEC for leading `(NIL)` */
00032 
00033 #define BOOST_DETAIL_PP_NILSEQ_IS_NIL(nilseq) \
00034     BOOST_PP_EQUAL(BOOST_DETAIL_PP_NILSEQ_SIZE(nilseq), 0)

00036 // Assume nilseq is not NIL and removes leading `(NIL)`.
00037 #define BOOST_DETAIL_PP_NILSEQ_TO_SEQ(nilseq) \
00038     BOOST_PP_SEQ_TAIL(nilseq)

00040 #define BOOST_DETAIL_PP_NILSEQ_ELEM(n, nilseq) \
00041     BOOST_PP_SEQ_ELEM(BOOST_PP_INC(n), nilseq) /* INC for leading `(NIL)` */
