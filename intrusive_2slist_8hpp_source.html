01310    template<class Predicate>
01311    void sort(Predicate p)
01312    {
01313       if (node_traits::get_next(node_traits::get_next(this->get_root_node()))
01314          != this->get_root_node()) {
01315 
01316          slist_impl carry(this->get_value_traits());
01317          detail::array_initializer<slist_impl, 64> counter(this->get_value_traits());
01318          int fill = 0;
01319          const_iterator last_inserted;
01320          while(!this->empty()){
01321             last_inserted = this->cbegin();
01322             carry.splice_after(carry.cbefore_begin(), *this, this->cbefore_begin());
01323             int i = 0;
01324             while(i < fill && !counter[i].empty()) {
01325                carry.swap(counter[i]);
01326                carry.merge(counter[i++], p, &last_inserted);
01327             }
01328             BOOST_INTRUSIVE_INVARIANT_ASSERT(counter[i].empty());
01329             const_iterator last_element(carry.previous(last_inserted, carry.end()));
01330 
01331             if(constant_time_size){
01332                counter[i].splice_after( counter[i].cbefore_begin(), carry
01333                                     , carry.cbefore_begin(), last_element
01334                                     , carry.size());
01335             }
01336             else{
01337                counter[i].splice_after( counter[i].cbefore_begin(), carry
01338                                     , carry.cbefore_begin(), last_element);
01339             }
01340             if(i == fill)
01341                ++fill;
01342          }
01343 
01344          for (int i = 1; i < fill; ++i)
01345             counter[i].merge(counter[i-1], p, &last_inserted);
01346          --fill;
01347          const_iterator last_element(counter[fill].previous(last_inserted, counter[fill].end()));
01348          if(constant_time_size){
01349             this->splice_after( cbefore_begin(), counter[fill], counter[fill].cbefore_begin()
01350                               , last_element, counter[fill].size());
01351          }
01352          else{
01353             this->splice_after( cbefore_begin(), counter[fill], counter[fill].cbefore_begin()
01354                               , last_element);
01355          }
01356       }
01357    }

01397    template<class Predicate>
01398    void merge(slist_impl& x, Predicate p, const_iterator *last = 0) 
01399    {
01400       const_iterator e(this->cend()), ex(x.cend()), bb(this->cbefore_begin()),
01401                      bb_next;
01402       if(last) *last = e.unconst();
01403       while(!x.empty()){
01404          const_iterator ibx_next(x.cbefore_begin()), ibx(ibx_next++);
01405          while (++(bb_next = bb) != e && !p(*ibx_next, *bb_next)){
01406             bb = bb_next;
01407          }
01408          if(bb_next == e){
01409             //Now transfer the rest to the end of the container
01410             this->splice_after(bb, x, last);
01411             break;
01412          }
01413          else{
01414             size_type n(0);
01415             do{
01416                ibx = ibx_next; ++n;
01417             } while(++(ibx_next = ibx) != ex && p(*ibx_next, *bb_next));
01418             this->splice_after(bb, x, x.before_begin(), ibx, n);
01419             if(last) *last = ibx;
01420          }
01421       }
01422    }

01849    static void priv_swap_cache_last(slist_impl *this_impl, slist_impl *other_impl)
01850    {
01851       bool other_was_empty = false;
01852       if(this_impl->empty()){
01853          //Check if both are empty or 
01854          if(other_impl->empty())
01855             return;
01856          //If this is empty swap pointers
01857          slist_impl *tmp = this_impl;
01858          this_impl  = other_impl;
01859          other_impl = tmp;
01860          other_was_empty = true;
01861       }
01862       else{
01863          other_was_empty = other_impl->empty();
01864       }
01865 
01866       //Precondition: this is not empty
01867       node_ptr other_old_last(other_impl->get_last_node());
01868       node_ptr other_bfirst(other_impl->get_root_node());
01869       node_ptr this_bfirst(this_impl->get_root_node());
01870       node_ptr this_old_last(this_impl->get_last_node());
01871 
01872       //Move all nodes from this to other's beginning
01873       node_algorithms::transfer_after(other_bfirst, this_bfirst, this_old_last);
01874       other_impl->set_last_node(this_old_last);
01875 
01876       if(other_was_empty){
01877          this_impl->set_last_node(this_bfirst);
01878       }
01879       else{
01880          //Move trailing nodes from other to this
01881          node_algorithms::transfer_after(this_bfirst, this_old_last, other_old_last);
01882          this_impl->set_last_node(other_old_last);
01883       }
01884    }
