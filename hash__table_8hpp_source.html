00127         inline std::size_t next_prime(std::size_t n) {
00128             std::size_t const* const prime_list_begin = prime_list::value;
00129             std::size_t const* const prime_list_end = prime_list_begin +
00130                 prime_list::length;
00131             std::size_t const* bound =
00132                 std::lower_bound(prime_list_begin, prime_list_end, n);
00133             if(bound == prime_list_end)
00134                 bound--;
00135             return *bound;

00231             // This copies the given function objects into the currently unused
00232             // function objects and returns a pointer, that func_ can later be
00233             // set to, to commit the change.
00234             //
00235             // Strong exception safety (since only usued function objects are
00236             // changed).
00237             functions_ptr buffer(buffered_functions const& x) {
00238                 functions_ptr ptr = func_ == &buffered_functions::func1_

00245                 BOOST_ASSERT(ptr != func_);

00251             }
