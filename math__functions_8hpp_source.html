00029 // gcd is an algorithm that calculates the greatest common divisor of two
00030 //  integers, using Euclid's algorithm.
00031 //
00032 // Pre: A > 0 && B > 0
00033 // Recommended: A > B
00034 template <typename Integer>
00035 inline Integer gcd(Integer A, Integer B)
00036 {
00037    do
00038    {
00039       const Integer tmp(B);
00040       B = A % B;
00041       A = tmp;
00042    } while (B != 0);
00043 
00044    return A;
00045 }

00047 //
00048 // lcm is an algorithm that calculates the least common multiple of two
00049 //  integers.
00050 //
00051 // Pre: A > 0 && B > 0
00052 // Recommended: A > B
00053 template <typename Integer>
00054 inline Integer lcm(const Integer & A, const Integer & B)
00055 {
00056    Integer ret = A;
00057    ret /= gcd(A, B);
00058    ret *= B;
00059    return ret;

00091    const bool Size_t_Bits_Power_2= !(Bits & (Bits-1));

00093 
00094    std::size_t n = x;
00095    std::size_t log2 = 0;
00096    
00097    for(std::size_t shift = Bits >> 1; shift; shift >>= 1){
00098       std::size_t tmp = n >> shift;
00099       if (tmp)
00100          log2 += shift, n = tmp;
