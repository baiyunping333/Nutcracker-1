00725    static void rebalance_after_erasure(node_ptr header, node_ptr x, node_ptr x_parent)
00726    {
00727       for (node_ptr root = NodeTraits::get_parent(header); x != root; root = NodeTraits::get_parent(header)) {
00728          const balance x_parent_balance = NodeTraits::get_balance(x_parent);
00729          if(x_parent_balance == NodeTraits::zero()){
00730             NodeTraits::set_balance(x_parent, 
00731                (x == NodeTraits::get_right(x_parent) ? NodeTraits::negative() : NodeTraits::positive()));
00732             break;       // the height didn't change, let's stop here
00733          }
00734          else if(x_parent_balance == NodeTraits::negative()){
00735             if (x == NodeTraits::get_left(x_parent)) {
00736                NodeTraits::set_balance(x_parent, NodeTraits::zero()); // balanced
00737                x = x_parent;
00738                x_parent = NodeTraits::get_parent(x_parent);
00739             }
00740             else {
00741                // x is right child
00742                // a is left child
00743                node_ptr a = NodeTraits::get_left(x_parent);
00744                BOOST_INTRUSIVE_INVARIANT_ASSERT(a);
00745                if (NodeTraits::get_balance(a) == NodeTraits::positive()) {
00746                   // a MUST have a right child
00747                   BOOST_INTRUSIVE_INVARIANT_ASSERT(NodeTraits::get_right(a));
00748                   rotate_left_right(x_parent, header);
00749                   x = NodeTraits::get_parent(x_parent);
00750                   x_parent = NodeTraits::get_parent(x);
00751                }
00752                else {
00753                   rotate_right(x_parent, header);
00754                   x = NodeTraits::get_parent(x_parent);
00755                   x_parent = NodeTraits::get_parent(x);
00756                }
00757 
00758                // if changed from negative to NodeTraits::positive(), no need to check above
00759                if (NodeTraits::get_balance(x) == NodeTraits::positive()){
00760                   break;
00761                }
00762             }
00763          }
00764          else if(x_parent_balance == NodeTraits::positive()){
00765             if (x == NodeTraits::get_right(x_parent)) {
00766                NodeTraits::set_balance(x_parent, NodeTraits::zero()); // balanced
00767                x = x_parent;
00768                x_parent = NodeTraits::get_parent(x_parent);
00769             }
00770             else {
00771                // x is left child
00772                // a is right child
00773                node_ptr a = NodeTraits::get_right(x_parent);
00774                BOOST_INTRUSIVE_INVARIANT_ASSERT(a);
00775                if (NodeTraits::get_balance(a) == NodeTraits::negative()) {
00776                   // a MUST have then a left child
00777                   BOOST_INTRUSIVE_INVARIANT_ASSERT(NodeTraits::get_left(a));
00778                   rotate_right_left(x_parent, header);
00779 
00780                   x = NodeTraits::get_parent(x_parent);
00781                   x_parent = NodeTraits::get_parent(x);
00782                }
00783                else {
00784                   rotate_left(x_parent, header);
00785                   x = NodeTraits::get_parent(x_parent);
00786                   x_parent = NodeTraits::get_parent(x);
00787                }
00788                // if changed from NodeTraits::positive() to negative, no need to check above
00789                if (NodeTraits::get_balance(x) == NodeTraits::negative()){
00790                   break;
00791                }
00792             }
00793          }
00794          else{
00795             BOOST_INTRUSIVE_INVARIANT_ASSERT(false);  // never reached
00796          }
00797       }
00798    }

00800    static void rebalance_after_insertion(node_ptr header, node_ptr x)
00801    {
00802       NodeTraits::set_balance(x, NodeTraits::zero());
00803 
00804       // Rebalance.
00805       for(node_ptr root = NodeTraits::get_parent(header); x != root; root = NodeTraits::get_parent(header)){
00806          const balance x_parent_balance = NodeTraits::get_balance(NodeTraits::get_parent(x));
00807 
00808          if(x_parent_balance == NodeTraits::zero()){
00809             // if x is left, parent will have parent->bal_factor = negative
00810             // else, parent->bal_factor = NodeTraits::positive()
00811             NodeTraits::set_balance( NodeTraits::get_parent(x)
00812                                     , x == NodeTraits::get_left(NodeTraits::get_parent(x))
00813                                        ? NodeTraits::negative() : NodeTraits::positive()  );
00814             x = NodeTraits::get_parent(x);
00815          }
00816          else if(x_parent_balance == NodeTraits::positive()){
00817             // if x is a left child, parent->bal_factor = zero
00818             if (x == NodeTraits::get_left(NodeTraits::get_parent(x)))
00819                NodeTraits::set_balance(NodeTraits::get_parent(x), NodeTraits::zero());
00820             else{        // x is a right child, needs rebalancing
00821                if (NodeTraits::get_balance(x) == NodeTraits::negative())
00822                   rotate_right_left(NodeTraits::get_parent(x), header);
00823                else
00824                   rotate_left(NodeTraits::get_parent(x), header);
00825             }
00826             break;
00827          }
00828          else if(x_parent_balance == NodeTraits::negative()){
00829             // if x is a left child, needs rebalancing
00830             if (x == NodeTraits::get_left(NodeTraits::get_parent(x))) {
00831                if (NodeTraits::get_balance(x) == NodeTraits::positive())
00832                   rotate_left_right(NodeTraits::get_parent(x), header);
00833                else
00834                   rotate_right(NodeTraits::get_parent(x), header);
00835             }
00836             else
00837                NodeTraits::set_balance(NodeTraits::get_parent(x), NodeTraits::zero());
00838             break;
00839          }
00840          else{
00841             BOOST_INTRUSIVE_INVARIANT_ASSERT(false);  // never reached
00842          }
00843       }
00844    }

00846    static void left_right_balancing(node_ptr a, node_ptr b, node_ptr c)
00847    {
00848       // balancing...
00849       const balance c_balance = NodeTraits::get_balance(c);
00850       const balance zero_balance = NodeTraits::zero();
00851       NodeTraits::set_balance(c, zero_balance);
00852       if(c_balance == NodeTraits::negative()){
00853          NodeTraits::set_balance(a, NodeTraits::positive());
00854          NodeTraits::set_balance(b, zero_balance);
00855       }
00856       else if(c_balance == zero_balance){
00857          NodeTraits::set_balance(a, zero_balance);
00858          NodeTraits::set_balance(b, zero_balance);
00859       }
00860       else if(c_balance == NodeTraits::positive()){
00861          NodeTraits::set_balance(a, zero_balance);
00862          NodeTraits::set_balance(b, NodeTraits::negative());
00863       }
00864       else{
00865          BOOST_INTRUSIVE_INVARIANT_ASSERT(false); // never reached

00869    static void rotate_left_right(const node_ptr a, node_ptr hdr)
00870    {
00871       //             |                               |         //
00872       //             a(-2)                           c         //
00873       //            / \                             / \        //
00874       //           /   \        ==>                /   \       //
00875       //      (pos)b    [g]                       b     a      //
00876       //          / \                            / \   / \     //
00877       //        [d]  c                         [d]  e f  [g]   //
00878       //           / \                                         //
00879       //          e   f                                        //
00880       node_ptr b = NodeTraits::get_left(a), c = NodeTraits::get_right(b);
00881       tree_algorithms::rotate_left(b, hdr);
00882       tree_algorithms::rotate_right(a, hdr);
00883       left_right_balancing(a, b, c);

00895       //             / \                                          //

00903    static void rotate_left(const node_ptr x, node_ptr hdr)
00904    {
00905       const node_ptr y = NodeTraits::get_right(x);
00906       tree_algorithms::rotate_left(x, hdr);
00907 
00908       // reset the balancing factor
00909       if (NodeTraits::get_balance(y) == NodeTraits::positive()) {
00910          NodeTraits::set_balance(x, NodeTraits::zero());
00911          NodeTraits::set_balance(y, NodeTraits::zero());
00912       }
00913       else {        // this doesn't happen during insertions
00914          NodeTraits::set_balance(x, NodeTraits::positive());
00915          NodeTraits::set_balance(y, NodeTraits::negative());
