00028 template<class Parent, class Member>
00029 inline std::ptrdiff_t offset_from_pointer_to_member(const Member Parent::* ptr_to_member)
00030 {
00031    //The implementation of a pointer to member is compiler dependent.
00032    #if defined(BOOST_INTRUSIVE_MSVC_COMPLIANT_PTR_TO_MEMBER)
00033    //msvc compliant compilers use their the first 32 bits as offset (even in 64 bit mode)
00034    return *(const boost::int32_t*)(void*)&ptr_to_member;
00035    //This works with gcc, msvc, ac++, ibmcpp
00036    #elif defined(__GNUC__)   || defined(__HP_aCC) || defined(BOOST_INTEL) || \
00037          defined(__IBMCPP__) || defined(__DECCXX)
00038    const Parent * const parent = 0;
00039    const char *const member = reinterpret_cast<const char*>(&(parent->*ptr_to_member));
00040    return std::ptrdiff_t(member - reinterpret_cast<const char*>(parent));
00041    #else
00042    //This is the traditional C-front approach: __MWERKS__, __DMC__, __SUNPRO_CC
00043    return (*(const std::ptrdiff_t*)(void*)&ptr_to_member) - 1;
00044    #endif
00045 }
