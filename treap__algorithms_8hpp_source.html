00806    static void rebalance_for_erasure(node_ptr header, node_ptr z, NodePtrPriorityCompare pcomp)
00807    {
00808       std::size_t n = 0;
00809       rerotate_on_destroy rb(header, z, n);
00810 
00811       node_ptr z_left  = NodeTraits::get_left(z);
00812       node_ptr z_right = NodeTraits::get_right(z);
00813       while(z_left || z_right){
00814          if(!z_right || (z_left && pcomp(z_left, z_right))){
00815             tree_algorithms::rotate_right(z, header);
00816          }
00817          else{
00818             tree_algorithms::rotate_left(z, header);
00819          }
00820          ++n;
00821          z_left  = NodeTraits::get_left(z);
00822          z_right = NodeTraits::get_right(z);
00823       }
00824       rb.release();
