00018 // Signature is represented as a seq during parsing (for efficient push-back
00019 // operations) but it is represented as an array after parsing (for efficient
00020 // random access of signature traits). These convert signature seq to array.
00021 // NOTE: The array is used instead of a typle because sometimes the random
00022 // access is needed before the signature has been fully parsed. In these cases
00023 // the total size is not known a-priory so the array conversion will just use
00024 // the number of sign elems parsed so far and still allow for random access.
00025 
00026 // NOTE: These macro intentionally do not check for errors. Make sure
00027 // seq_sign_err is validated before it is passed to these macros.
00028 
00029 // Convert and return just sign. Err should be 0.
00030 #define BOOST_DETAIL_PP_SIGN_PARSE_FINISH_SIGN_(seq_sign_err) \
00031     BOOST_PP_SEQ_TO_ARRAY(BOOST_PP_SEQ_TAIL( \
00032             BOOST_DETAIL_PP_SIGN_PARSER_SIGN(seq_sign_err) \
00033             /* terminate array with NIL to avoid EMPTY terminated arrays */ \
00034             (BOOST_PP_NIL) \
00035             ))
00036 
00037 // Convert sign and return (seq, converted-sign, err). If err != 0 it will be
00038 // returned as such but sign convertion will still be done.
00039 #define BOOST_DETAIL_PP_SIGN_PARSE_FINISH(seq_sign_err) \
00040     ( \
00041         BOOST_DETAIL_PP_SIGN_PARSER_SEQ(seq_sign_err) \
00042     , \
00043         BOOST_DETAIL_PP_SIGN_PARSE_FINISH_SIGN_(seq_sign_err) \
