00735    template<class H_Alpha>
00736    static void rebalance_after_insertion
00737       ( node_ptr x, std::size_t depth
00738       , std::size_t tree_size, H_Alpha h_alpha, std::size_t &max_tree_size)
00739    {
00740       if(tree_size > max_tree_size)
00741          max_tree_size = tree_size;
00742 
00743       if(tree_size != 1 && depth > h_alpha(tree_size)){
00744          //Find the first non height-balanced node
00745          //as described in the section 4.2 of the paper.
00746          //This method is the alternative method described
00747          //in the paper. Authors claim that this method
00748          //may tend to yield more balanced trees on the average
00749          //than the weight balanced method.
00750          node_ptr s = x;
00751          std::size_t size = 1;
00752 
00753          for(std::size_t i = 1; true; ++i){
00754             bool rebalance = false;
00755             if(i == depth){
00756                BOOST_INTRUSIVE_INVARIANT_ASSERT(tree_size == count(s));
00757                rebalance = true;
00758             }
00759             else if(i > h_alpha(size)){
00760                node_ptr s_parent = NodeTraits::get_parent(s);
00761                node_ptr s_parent_left = NodeTraits::get_left(s_parent);
00762                size += 1 + tree_algorithms::count
00763                   ( s_parent_left == s ? NodeTraits::get_right(s_parent) : s_parent_left );
00764                s = s_parent;
00765                rebalance = true;
00766             }
00767             if(rebalance){
00768                rebalance_subtree(s);
00769                break;
00770             }
00771          }
00772       }
00773    }
