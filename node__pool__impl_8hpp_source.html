00094    multiallocation_chain allocate_nodes(const std::size_t n)
00095    {
00096       //Preallocate all needed blocks to fulfill the request
00097       std::size_t cur_nodes = m_freelist.size();
00098       if(cur_nodes < n){
00099          priv_alloc_block(((n - cur_nodes) - 1)/m_nodes_per_block + 1);
00100       }
00101 
00102       //We just iterate the needed nodes to get the last we'll erase
00103       typedef typename free_nodes_t::iterator free_iterator;
00104       free_iterator before_last_new_it = m_freelist.before_begin();
00105       for(std::size_t j = 0; j != n; ++j){
00106          ++before_last_new_it;
00107       }
00108 
00109       //Cache the first node of the allocated range before erasing
00110       free_iterator first_node(m_freelist.begin());
00111       free_iterator last_node (before_last_new_it);
00112 
00113       //Erase the range. Since we already have the distance, this is O(1)
00114       m_freelist.erase_after( m_freelist.before_begin()
00115                             , ++free_iterator(before_last_new_it)
00116                             , n);
00117 
00118       //Now take the last erased node and just splice it in the end
00119       //of the intrusive list that will be traversed by the multialloc iterator.
00120       multiallocation_chain chain;
00121       chain.incorporate_after(chain.before_begin(), &*first_node, &*last_node, n);
00122       m_allocated += n;
00123       return BOOST_CONTAINER_MOVE_NAMESPACE::move(chain);
00124    }

00138    void deallocate_free_blocks()
00139    {
00140       typedef typename free_nodes_t::iterator nodelist_iterator;
00141       typename blockslist_t::iterator bit(m_blocklist.before_begin()),
00142                                       it(m_blocklist.begin()),
00143                                       itend(m_blocklist.end());
00144       free_nodes_t backup_list;
00145       nodelist_iterator backup_list_last = backup_list.before_begin();
00146 
00147       //Execute the algorithm and get an iterator to the last value
00148       std::size_t blocksize = get_rounded_size
00149          (m_real_node_size*m_nodes_per_block, alignment_of<node_t>::value);
00150 
00151       while(it != itend){
00152          //Collect all the nodes from the block pointed by it
00153          //and push them in the list
00154          free_nodes_t free_nodes;
00155          nodelist_iterator last_it = free_nodes.before_begin();
00156          const void *addr = get_block_from_hook(&*it, blocksize);
00157 
00158          m_freelist.remove_and_dispose_if
00159             (is_between(addr, blocksize), push_in_list(free_nodes, last_it));
00160 
00161          //If the number of nodes is equal to m_nodes_per_block
00162          //this means that the block can be deallocated
00163          if(free_nodes.size() == m_nodes_per_block){
00164             //Unlink the nodes
00165             free_nodes.clear();
00166             it = m_blocklist.erase_after(bit);
00167             mp_segment_mngr_base->deallocate((void*)addr);
00168          }
00169          //Otherwise, insert them in the backup list, since the
00170          //next "remove_if" does not need to check them again.
00171          else{
00172             //Assign the iterator to the last value if necessary
00173             if(backup_list.empty() && !m_freelist.empty()){
00174                backup_list_last = last_it;
00175             }
00176             //Transfer nodes. This is constant time.
00177             backup_list.splice_after
00178                ( backup_list.before_begin()
00179                , free_nodes
00180                , free_nodes.before_begin()
00181                , last_it
00182                , free_nodes.size());
00183             bit = it;
00184             ++it;
00185          }
00186       }
00187       //We should have removed all the nodes from the free list
00188       assert(m_freelist.empty());
00189 
00190       //Now pass all the node to the free list again
00191       m_freelist.splice_after
00192          ( m_freelist.before_begin()
00193          , backup_list
00194          , backup_list.before_begin()
00195          , backup_list_last
00196          , backup_list.size());
00197    }

00297    void priv_alloc_block(std::size_t num_blocks = 1)
00298    {
00299       if(!num_blocks)
00300          return;
00301       std::size_t blocksize = 
00302          get_rounded_size(m_real_node_size*m_nodes_per_block, alignment_of<node_t>::value);
00303 
00304       try{
00305          for(std::size_t i = 0; i != num_blocks; ++i){
00306             //We allocate a new NodeBlock and put it as first
00307             //element in the free Node list
00308             char *pNode = reinterpret_cast<char*>
00309                (mp_segment_mngr_base->allocate(blocksize + sizeof(node_t)));
00310             char *pBlock = pNode;
00311             m_blocklist.push_front(get_block_hook(pBlock, blocksize));
00312 
00313             //We initialize all Nodes in Node Block to insert 
00314             //them in the free Node list
00315             for(std::size_t i = 0; i < m_nodes_per_block; ++i, pNode += m_real_node_size){
00316                m_freelist.push_front(*new (pNode) node_t);
00317             }
00318          }
00319       }
00320       catch(...){
00321          //to-do: if possible, an efficient way to deallocate allocated blocks
00322          throw;
00323       }
00324    }
