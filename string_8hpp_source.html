01679    template <class ForwardIter>
01680    void priv_insert(iterator position, ForwardIter first, 
01681                     ForwardIter last,  std::forward_iterator_tag)
01682    {
01683       if (first != last) {
01684          size_type n = std::distance(first, last);
01685          size_type remaining = this->capacity() - this->priv_size();
01686          const size_type old_size = this->size();
01687          pointer old_start = this->priv_addr();
01688          bool enough_capacity = false;
01689          std::pair<pointer, bool> allocation_ret;
01690          size_type new_cap = 0;
01691 
01692          //Check if we have enough capacity
01693          if (remaining >= n){
01694             enough_capacity = true;            
01695          }
01696          else {
01697             //Otherwise expand current buffer or allocate new storage
01698             new_cap  = this->next_capacity(n);
01699             allocation_ret = this->allocation_command
01700                   (allocate_new | expand_fwd | expand_bwd, old_size + n + 1, 
01701                      new_cap, new_cap, old_start);
01702 
01703             //Check forward expansion
01704             if(old_start == allocation_ret.first){
01705                enough_capacity = true;
01706                this->priv_storage(new_cap);
01707             }
01708          }
01709 
01710          //Reuse same buffer
01711          if(enough_capacity){
01712             const size_type elems_after =
01713                this->priv_size() - (position - this->priv_addr());
01714             size_type old_length = this->priv_size();
01715             if (elems_after >= n) {
01716                pointer pointer_past_last = this->priv_addr() + this->priv_size() + 1;
01717                priv_uninitialized_copy(this->priv_addr() + (this->priv_size() - n + 1),
01718                                        pointer_past_last, pointer_past_last);
01719 
01720                this->priv_size(this->priv_size()+n);
01721                Traits::move(containers_detail::get_pointer(position + n),
01722                            containers_detail::get_pointer(position),
01723                            (elems_after - n) + 1);
01724                this->priv_copy(first, last, position);
01725             }
01726             else {
01727                ForwardIter mid = first;
01728                std::advance(mid, elems_after + 1);
01729 
01730                priv_uninitialized_copy(mid, last, this->priv_addr() + this->priv_size() + 1);
01731                this->priv_size(this->priv_size() + (n - elems_after));
01732                priv_uninitialized_copy
01733                   (position, this->priv_addr() + old_length + 1, 
01734                   this->priv_addr() + this->priv_size());
01735                this->priv_size(this->priv_size() + elems_after);
01736                this->priv_copy(first, mid, position);
01737             }
01738          }
01739          else{
01740             pointer new_start = allocation_ret.first;
01741             if(!allocation_ret.second){
01742                //Copy data to new buffer
01743                size_type new_length = 0;
01744                //This can't throw, since characters are POD
01745                new_length += priv_uninitialized_copy
01746                               (this->priv_addr(), position, new_start);
01747                new_length += priv_uninitialized_copy
01748                               (first, last, new_start + new_length);
01749                new_length += priv_uninitialized_copy
01750                               (position, this->priv_addr() + this->priv_size(), 
01751                               new_start + new_length);
01752                this->priv_construct_null(new_start + new_length);
01753 
01754                this->deallocate_block();
01755                this->is_short(false);
01756                this->priv_addr(new_start);
01757                this->priv_size(new_length);
01758                this->priv_storage(new_cap);
01759             }
01760             else{
01761                //value_type is POD, so backwards expansion is much easier 
01762                //than with vector<T>
01763                value_type *oldbuf = containers_detail::get_pointer(old_start);
01764                value_type *newbuf = containers_detail::get_pointer(new_start);
01765                value_type *pos    = containers_detail::get_pointer(position);
01766                size_type  before  = pos - oldbuf;
01767 
01768                //First move old data
01769                Traits::move(newbuf, oldbuf, before);
01770                Traits::move(newbuf + before + n, pos, old_size - before);
01771                //Now initialize the new data
01772                priv_uninitialized_copy(first, last, new_start + before);
01773                this->priv_construct_null(new_start + (old_size + n));
01774                this->is_short(false);
01775                this->priv_addr(new_start);
01776                this->priv_size(old_size + n);
01777                this->priv_storage(new_cap);
01778             }
01779          }
01780       }
01781    }
