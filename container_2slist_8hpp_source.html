00924    void resize(size_type new_size, const T& x)
00925    {
00926       typename Icont::iterator end_n(this->icont().end()), cur(this->icont().before_begin()), cur_next;
00927       while (++(cur_next = cur) != end_n && new_size > 0){
00928          --new_size;
00929          cur = cur_next;
00930       }
00931       if (cur_next != end_n) 
00932          this->erase_after(const_iterator(cur), const_iterator(end_n));
00933       else
00934          this->insert_after(const_iterator(cur), new_size, x);
00935    }
