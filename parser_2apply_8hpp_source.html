00023 // Apply specified operation macro: If no error then expand to
00024 // `op_macro(sign, data)`, else to `err_macro(message, data)` (where
00025 // `message` is a valid concatenable pp single token like `ERROR_abc_xyz`).
00026 // This can be used to process parsed signature via `op_macro()` while ensuring
00027 // parsing macro (usually computationally expensive) is expanded only once:
00028 //     #define PARSE_(seq) ... // Parse `seq` into `sign_seq_err`.
00029 //     #define CODE_(sign, data) ... // Generate code from parsed signature.
00030 //     #define ERR_(msg, data) ; BOOST_MPL_ASSERT_MSG(false, msg, ()) ;
00031 //     #define CODE(seq) PP_SIGN_PARSE_APPLY(PARSE_(seq), CODE_, ERR_, ~)
00032 #define BOOST_DETAIL_PP_SIGN_PARSER_APPLY( \
00033         seq_sign_err, op_macro, err_macro, data) \
00034     BOOST_PP_IIF(BOOST_DETAIL_PP_SIGN_PARSER_HAS_ERR(seq_sign_err), \
00035         BOOST_DETAIL_PP_SIGN_PARSER_APPLY_ERR_ /* expand to error */ \
00036     , /* else */ \
00037         BOOST_DETAIL_PP_SIGN_PARSER_APPLY_OK_ /* expand to code */ \
00038     )(seq_sign_err, op_macro, err_macro, data)
