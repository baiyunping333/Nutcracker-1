01266    void priv_range_insert(pointer pos, const size_type n, advanced_insert_aux_int_t &interf)
01267    {
01268       //Check if we have enough memory or try to expand current memory
01269       size_type remaining = this->members_.m_capacity - this->members_.m_size;
01270       bool same_buffer_start;
01271       std::pair<pointer, bool> ret;
01272       size_type real_cap = this->members_.m_capacity;
01273 
01274       //Check if we already have room
01275       if (n <= remaining){
01276          same_buffer_start = true;
01277       }
01278       else{
01279          //There is not enough memory, allocate a new
01280          //buffer or expand the old one.
01281          size_type new_cap = this->next_capacity(n);
01282          ret = this->allocation_command
01283                (allocate_new | expand_fwd | expand_bwd,
01284                   this->members_.m_size + n, new_cap, real_cap, this->members_.m_start);
01285 
01286          //Check for forward expansion
01287          same_buffer_start = ret.second && this->members_.m_start == ret.first;
01288          if(same_buffer_start){
01289             this->members_.m_capacity  = real_cap;
01290          }
01291       }
01292       
01293       //If we had room or we have expanded forward
01294       if (same_buffer_start){
01295          #ifdef BOOST_CONTAINERS_VECTOR_ALLOC_STATS
01296          ++this->num_expand_fwd;
01297          #endif
01298          this->priv_range_insert_expand_forward
01299             (containers_detail::get_pointer(pos), n, interf);
01300       }
01301       //Backwards (and possibly forward) expansion
01302       else if(ret.second){
01303          #ifdef BOOST_CONTAINERS_VECTOR_ALLOC_STATS
01304          ++this->num_expand_bwd;
01305          #endif
01306          this->priv_range_insert_expand_backwards
01307             ( containers_detail::get_pointer(ret.first)
01308             , real_cap
01309             , containers_detail::get_pointer(pos)
01310             , n
01311             , interf);
01312       }
01313       //New buffer
01314       else{
01315          #ifdef BOOST_CONTAINERS_VECTOR_ALLOC_STATS
01316          ++this->num_alloc;
01317          #endif
01318          this->priv_range_insert_new_allocation
01319             ( containers_detail::get_pointer(ret.first)
01320             , real_cap
01321             , containers_detail::get_pointer(pos)
01322             , n
01323             , interf);
01324       }
01325    }

01327    void priv_range_insert_expand_forward(T* pos, size_type n, advanced_insert_aux_int_t &interf)
01328    {
01329       //n can't be 0, because there is nothing to do in that case
01330       if(!n) return;
01331       //There is enough memory
01332       T* old_finish = containers_detail::get_pointer(this->members_.m_start) + this->members_.m_size;
01333       const size_type elems_after = old_finish - pos;
01334 
01335       if (elems_after > n){
01336          //New elements can be just copied.
01337          //Move to uninitialized memory last objects
01338          ::BOOST_CONTAINER_MOVE_NAMESPACE::uninitialized_move(old_finish - n, old_finish, old_finish);
01339          this->members_.m_size += n;
01340          //Copy previous to last objects to the initialized end
01341          BOOST_CONTAINER_MOVE_NAMESPACE::move_backward(pos, old_finish - n, old_finish);
01342          //Insert new objects in the pos
01343          interf.copy_all_to(pos);
01344       }
01345       else {
01346          //The new elements don't fit in the [pos, end()) range. Copy
01347          //to the beginning of the unallocated zone the last new elements.
01348          interf.uninitialized_copy_some_and_update(old_finish, elems_after, false);
01349          this->members_.m_size += n - elems_after;
01350          //Copy old [pos, end()) elements to the uninitialized memory
01351          ::BOOST_CONTAINER_MOVE_NAMESPACE::uninitialized_move
01352             ( pos, old_finish, containers_detail::get_pointer(this->members_.m_start) + this->members_.m_size);
01353          this->members_.m_size += elems_after;
01354          //Copy first new elements in pos
01355          interf.copy_all_to(pos);
01356       }
01357    }

01359    void priv_range_insert_new_allocation
01360       (T* new_start, size_type new_cap, T* pos, size_type n, advanced_insert_aux_int_t &interf)
01361    {
01362       //n can be zero, if we want to reallocate!
01363       T *new_finish = new_start;
01364       T *old_finish;
01365       //Anti-exception rollbacks
01366       typename value_traits::UCopiedArrayDeallocator scoped_alloc(new_start, this->alloc(), new_cap);
01367       typename value_traits::UCopiedArrayDestructor constructed_values_destroyer(new_start, 0u);
01368 
01369       //Initialize with [begin(), pos) old buffer 
01370       //the start of the new buffer
01371       T *old_buffer = containers_detail::get_pointer(this->members_.m_start);
01372       if(old_buffer){
01373          new_finish = ::BOOST_CONTAINER_MOVE_NAMESPACE::uninitialized_move
01374             (containers_detail::get_pointer(this->members_.m_start), pos, old_finish = new_finish);
01375          constructed_values_destroyer.increment_size(new_finish - old_finish);
01376       }
01377       //Initialize new objects, starting from previous point
01378       interf.uninitialized_copy_all_to(old_finish = new_finish);
01379       new_finish += n;
01380       constructed_values_destroyer.increment_size(new_finish - old_finish);
01381       //Initialize from the rest of the old buffer, 
01382       //starting from previous point
01383       if(old_buffer){
01384          new_finish = ::BOOST_CONTAINER_MOVE_NAMESPACE::uninitialized_move
01385             (pos, old_buffer + this->members_.m_size, new_finish);
01386          //Destroy and deallocate old elements
01387          //If there is allocated memory, destroy and deallocate
01388          if(!value_traits::trivial_dctr_after_move)
01389             this->destroy_n(old_buffer, this->members_.m_size); 
01390          this->alloc().deallocate(this->members_.m_start, this->members_.m_capacity);
01391       }
01392       this->members_.m_start     = new_start;
01393       this->members_.m_size      = new_finish - new_start;
01394       this->members_.m_capacity  = new_cap;
01395       //All construction successful, disable rollbacks
01396       constructed_values_destroyer.release();
01397       scoped_alloc.release();
01398    }

01400    void priv_range_insert_expand_backwards
01401          (T* new_start, size_type new_capacity,
01402           T* pos, const size_type n, advanced_insert_aux_int_t &interf)
01403    {
01404       //n can be zero to just expand capacity
01405       //Backup old data
01406       T* old_start  = containers_detail::get_pointer(this->members_.m_start);
01407       T* old_finish = old_start + this->members_.m_size;
01408       size_type old_size = this->members_.m_size;
01409 
01410       //We can have 8 possibilities:
01411       const size_type elemsbefore   = (size_type)(pos - old_start);
01412       const size_type s_before      = (size_type)(old_start - new_start);
01413 
01414       //Update the vector buffer information to a safe state
01415       this->members_.m_start      = new_start;
01416       this->members_.m_capacity   = new_capacity;
01417       this->members_.m_size = 0;
01418 
01419       //If anything goes wrong, this object will destroy
01420       //all the old objects to fulfill previous vector state
01421       typename value_traits::OldArrayDestructor old_values_destroyer(old_start, old_size);
01422       //Check if s_before is big enough to hold the beginning of old data + new data
01423       if(difference_type(s_before) >= difference_type(elemsbefore + n)){
01424          //Copy first old values before pos, after that the new objects
01425          ::BOOST_CONTAINER_MOVE_NAMESPACE::uninitialized_move(old_start, pos, new_start);
01426          this->members_.m_size = elemsbefore;
01427          interf.uninitialized_copy_all_to(new_start + elemsbefore);
01428          this->members_.m_size += n;
01429          //Check if s_before is so big that even copying the old data + new data
01430          //there is a gap between the new data and the old data
01431          if(s_before >= (old_size + n)){
01432             //Old situation:
01433             // _________________________________________________________
01434             //|            raw_mem                | old_begin | old_end |
01435             //| __________________________________|___________|_________|
01436             //
01437             //New situation:
01438             // _________________________________________________________
01439             //| old_begin |    new   | old_end |         raw_mem        |
01440             //|___________|__________|_________|________________________|
01441             //
01442             //Now initialize the rest of memory with the last old values
01443             ::BOOST_CONTAINER_MOVE_NAMESPACE::uninitialized_move
01444                (pos, old_finish, new_start + elemsbefore + n);
01445             //All new elements correctly constructed, avoid new element destruction
01446             this->members_.m_size = old_size + n;
01447             //Old values destroyed automatically with "old_values_destroyer"
01448             //when "old_values_destroyer" goes out of scope unless the have trivial
01449             //destructor after move.
01450             if(value_traits::trivial_dctr_after_move)
01451                old_values_destroyer.release();
01452          }
01453          //s_before is so big that divides old_end
01454          else{
01455             //Old situation:
01456             // __________________________________________________
01457             //|            raw_mem         | old_begin | old_end |
01458             //| ___________________________|___________|_________|
01459             //
01460             //New situation:
01461             // __________________________________________________
01462             //| old_begin |   new    | old_end |  raw_mem        |
01463             //|___________|__________|_________|_________________|
01464             //
01465             //Now initialize the rest of memory with the last old values
01466             //All new elements correctly constructed, avoid new element destruction
01467             size_type raw_gap = s_before - (elemsbefore + n);
01468             //Now initialize the rest of s_before memory with the
01469             //first of elements after new values
01470             ::BOOST_CONTAINER_MOVE_NAMESPACE::uninitialized_move(pos, pos + raw_gap, new_start + elemsbefore + n);
01471             //Update size since we have a contiguous buffer
01472             this->members_.m_size = old_size + s_before;
01473             //All new elements correctly constructed, avoid old element destruction
01474             old_values_destroyer.release();
01475             //Now copy remaining last objects in the old buffer begin
01476             T *to_destroy = BOOST_CONTAINER_MOVE_NAMESPACE::move(pos + raw_gap, old_finish, old_start);
01477             //Now destroy redundant elements except if they were moved and
01478             //they have trivial destructor after move
01479             size_type n_destroy =  old_finish - to_destroy;
01480             if(!value_traits::trivial_dctr_after_move)
01481                this->destroy_n(to_destroy, n_destroy);
01482             this->members_.m_size -= n_destroy;
01483          }
01484       }
01485       else{
01486          //Check if we have to do the insertion in two phases
01487          //since maybe s_before is not big enough and
01488          //the buffer was expanded both sides
01489          //
01490          //Old situation:
01491          // _________________________________________________
01492          //| raw_mem | old_begin + old_end |  raw_mem        |
01493          //|_________|_____________________|_________________|
01494          //
01495          //New situation with do_after:
01496          // _________________________________________________
01497          //|     old_begin + new + old_end     |  raw_mem    |
01498          //|___________________________________|_____________|
01499          //
01500          //New without do_after:
01501          // _________________________________________________
01502          //| old_begin + new + old_end  |  raw_mem           |
01503          //|____________________________|____________________|
01504          //
01505          bool do_after    = n > s_before;
01506 
01507          //Now we can have two situations: the raw_mem of the
